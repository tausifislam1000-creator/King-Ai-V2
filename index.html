export default async function handler(req, res) { // Only allow POST requests if (req.method !== 'POST') { return res.status(405).json({ error: 'Method Not Allowed' }); } const SECRET_KEY = process.env.OPENROUTER_API_KEY; try { const response = await fetch('https://openrouter.ai/api/v1/chat/completions', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${SECRET_KEY}`, // It's good practice to include these headers for OpenRouter 'HTTP-Referer': 'https://yourwebsite.com', 'X-Title': 'King AI' }, body: JSON.stringify(req.body) }); const data = await response.json(); res.status(200).json(data); } catch (error) { console.error("OpenRouter Error:", error); res.status(500).json({ error: 'Internal Server Error' }); } }
